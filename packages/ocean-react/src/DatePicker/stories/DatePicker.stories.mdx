# DatePicker APIs

The API documentation of the DatePicker React component. Learn more about the props and the CSS customization points.

## Import

```javascript
import { DatePicker } from '@useblu/ocean-react';
```

export const DinamicDatePicker = () => {
  const labels = {
    from: 'Start date',
    to: 'End date',
  };
  const [dates, setDates] = useState({ from: '', to: '' });
  const handleSelectDates = (dates) =>
    setDates({ from: dates.from, to: dates.to });
  return (
    <div style={{ height: '350px' }}>
      <DatePicker
        labels={labels}
        values={dates}
        onSelect={handleSelectDates}
        editable
      />
    </div>
  );
};

## Usage

```jsx
import { useState } from 'react';
import { DatePicker } from '@useblu/ocean-react';

const [dates, setDates] = useState({ from: '', to: '' });

const handleSelectDates = (dates) =>
  setDate({ from: dates.from, to: dates.to });

const labels = {
  from: 'Start date',
  to: 'End date',
};

<DatePicker labels={labels} values={dates} onSelect={handleSelectDates} />;
```

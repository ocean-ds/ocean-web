import { useState } from 'react';

  title="Components/DatePicker/Playground"

export const DinamicDatePicker = () => {
  const labels = {
    from: 'Start date',
    to: 'End date',
  };
  const [dates, setDates] = useState({ from: '', to: '' });
  const handleSelectDates = (dates) =>
    setDates({ from: dates.from, to: dates.to });
  return (
    <div
      style={{
        width: '100%',
        height: '400px',
        display: 'flex',
        justifyContent: 'center',
      }}
    >
      <div style={{ width: '100%' }}>
        <DatePicker
          labels={labels}
          values={dates}
          onSelect={handleSelectDates}
          editable
        />
      </div>
    </div>
  );
};
